37      @PostMapping("/{propertyName}")
38      @ResponseStatus(HttpStatus.CREATED)
39      public ResponseEntity<Object> create(@PathVariable String propertyName, @RequestBody Booking booking) throws Exception {
40          log.info("Creating booking: {}", booking);
41          val maybePropertyEntity = propertyDao.findByName(propertyName);
42          //todo grab email from the security context so this can't fail
43          val customerEntity = customerDao.findByEmail(booking.getCustomer().getEmail());
44  
45          final Function<Booking, URI> toBookingUri = b -> {
46              try {
47                  return new URI("/bookingId");
48              } catch (URISyntaxException e) {
49                  throw new RuntimeException(e);
50              }
51          };
52  
53          return maybePropertyEntity
54              .map(propertyEntity -> new BookingEntity(booking.getDateRange().toEmbeddable(), customerEntity, propertyEntity))
55              .map(bookingDao::save)
56              .map(Booking::from)
57              .map(toBookingUri)
58              .map(uri -> ResponseEntity.created(uri).build())
59              .orElse(ResponseEntity.badRequest().body(Failure.PROPERTY_NOT_FOUND.toString()));
60      }

62      @GetMapping
